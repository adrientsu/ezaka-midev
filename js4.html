🏷️ Classes et Programmation Orientée Objet

Créer des modèles d'objets réutilisables

🔍 16.1 Qu'est-ce que la Programmation Orientée Objet ?

La Programmation Orientée Objet (POO) est un paradigme qui organise le code autour d'objets qui contiennent des données (propriétés) et des comportements (méthodes).

📖 Concepts clés :

  • Classe : modèle/plan pour créer des objets
  • Objet : instance d'une classe
  • Constructeur : méthode appelée à la création
  • Héritage : une classe peut hériter d'une autre
  • Encapsulation : cacher les détails internes

🏗️ 16.2 Créer une classe (ES6)

// Déclaration de classe
class Personne {
    // Constructeur (appelé avec new)
    constructor(nom, age) {
        this.nom = nom;
        this.age = age;
    }
    
    // Méthode
    sePresenter() {
        console.log(`Bonjour, je m'appelle ${this.nom} et j'ai ${this.age} ans.`);
    }
    
    // Autre méthode
    anniversaire() {
        this.age++;
        console.log(`Joyeux anniversaire ${this.nom} ! Vous avez maintenant ${this.age} ans.`);
    }
}

// Créer une instance
const personne1 = new Personne("Jean", 25);
const personne2 = new Personne("Marie", 30);

personne1.sePresenter(); // "Bonjour, je m'appelle Jean et j'ai 25 ans."
personne2.sePresenter(); // "Bonjour, je m'appelle Marie et j'ai 30 ans."
personne1.anniversaire(); // "Joyeux anniversaire Jean ! Vous avez maintenant 26 ans."
Note : Par convention, les noms de classes commencent par une majuscule (PascalCase).

⚙️ 16.3 Propriétés et méthodes avancées

class CompteBancaire {
    constructor(titulaire, soldeInitial = 0) {
        this.titulaire = titulaire;
        this._solde = soldeInitial; // Convention : _ pour "privé"
    }
    
    // Getter
    get solde() {
        return this._solde;
    }
    
    // Setter
    set solde(montant) {
        if (montant < 0) {
            console.error("Le solde ne peut pas être négatif");
            return;
        }
        this._solde = montant;
    }
    
    // Méthode
    deposer(montant) {
        if (montant <= 0) {
            console.error("Le montant doit être positif");
            return;
        }
        this._solde += montant;
        console.log(`Dépôt de ${montant}€. Nouveau solde: ${this._solde}€`);
    }
    
    retirer(montant) {
        if (montant <= 0) {
            console.error("Le montant doit être positif");
            return;
        }
        if (montant > this._solde) {
            console.error("Solde insuffisant");
            return;
        }
        this._solde -= montant;
        console.log(`Retrait de ${montant}€. Nouveau solde: ${this._solde}€`);
    }
    
    // Méthode statique (appelée sur la classe, pas sur l'instance)
    static convertirEnEuro(montant, devise) {
        const taux = { USD: 0.92, GBP: 1.17, MGA: 0.00022 };
        return montant * (taux[devise] || 1);
    }
}

// Utilisation
const compte = new CompteBancaire("Jean", 1000);
console.log(compte.solde); // 1000 (getter)
compte.deposer(500);        // Dépôt de 500€. Nouveau solde: 1500€
compte.retirer(200);        // Retrait de 200€. Nouveau solde: 1300€

// Méthode statique
console.log(CompteBancaire.convertirEnEuro(100, "USD")); // 92

👪 16.4 Héritage (extends)

Une classe peut hériter d'une autre classe avec extends.

class Animal {
    constructor(nom) {
        this.nom = nom;
    }
    
    parler() {
        console.log(`${this.nom} fait un bruit.`);
    }
}

class Chien extends Animal {
    constructor(nom, race) {
        super(nom); // Appelle le constructeur parent
        this.race = race;
    }
    
    // Redéfinition de méthode (override)
    parler() {
        console.log(`${this.nom} (${this.race}) aboie: Ouaf !`);
    }
    
    // Nouvelle méthode spécifique
    rapporter() {
        console.log(`${this.nom} rapporte la balle !`);
    }
}

class Chat extends Animal {
    parler() {
        console.log(`${this.nom} miaule: Miaou !`);
    }
}

const rex = new Chien("Rex", "Berger Allemand");
const felix = new Chat("Félix");

rex.parler();    // "Rex (Berger Allemand) aboie: Ouaf !"
felix.parler();  // "Félix miaule: Miaou !"
rex.rapporter(); // "Rex rapporte la balle !"
// felix.rapporter(); // Erreur !

🔒 16.5 Propriétés privées (#)

Depuis ES2020, on peut utiliser # pour des propriétés vraiment privées.

class Banque {
    // Propriété privée
    #solde;
    
    constructor(titulaire, soldeInitial) {
        this.titulaire = titulaire;
        this.#solde = soldeInitial;
    }
    
    get solde() {
        return this.#solde;
    }
    
    deposer(montant) {
        if (montant > 0) {
            this.#solde += montant;
        }
    }
    
    // Impossible d'accéder à #solde de l'extérieur
}

const compte = new Banque("Jean", 1000);
console.log(compte.solde);     // 1000 (via getter)
// console.log(compte.#solde); // Erreur ! (privé)

📦 Modules JavaScript

Organiser et partager du code

🔍 17.1 Pourquoi les modules ?

Les modules permettent de diviser le code en fichiers séparés, chacun avec sa propre portée. On peut exporter des parties d'un module et les importer dans d'autres.

💡
Avantages :
  • Code plus organisé
  • Réutilisable
  • Évite les conflits de noms
  • Meilleure maintenabilité

📤 17.2 export et import

Fichier math.js (export)

// Export nommé
export const PI = 3.14159;

export function addition(a, b) {
    return a + b;
}

export function multiplication(a, b) {
    return a * b;
}

// Export par défaut (un seul par module)
export default function saluer(nom) {
    return `Bonjour ${nom} !`;
}

Fichier app.js (import)

// Importer des exports nommés
import { addition, multiplication, PI } from './math.js';

console.log(addition(5, 3));      // 8
console.log(multiplication(4, 2)); // 8
console.log(PI);                   // 3.14159

// Importer l'export par défaut
import saluer from './math.js';
console.log(saluer("Jean"));       // "Bonjour Jean !"

// Importer avec alias
import { addition as add, multiplication as mult } from './math.js';
console.log(add(10, 5));           // 15

// Tout importer sous un namespace
import * as Maths from './math.js';
console.log(Maths.addition(7, 2)); // 9

🌐 17.3 Utiliser des modules dans le navigateur

<!-- Dans le HTML -->
<script type="module" src="app.js"></script>

<!-- Ou directement dans le HTML -->
<script type="module">
    import { addition } from './math.js';
    console.log(addition(5, 3));
</script>
⚠️
Attention : Les modules CORS (Cross-Origin Resource Sharing) nécessitent un serveur. Vous ne pouvez pas les tester directement avec le protocole file://. Utilisez Live Server ou un serveur local.

⚠️ Gestion des erreurs

try, catch, finally, throw

🔍 18.1 try/catch

Le bloc try/catch permet de gérer les erreurs sans interrompre l'exécution du programme.

try {
    // Code qui peut générer une erreur
    let resultat = 10 / 0;
    console.log("Résultat:", resultat);
    
    // Erreur volontaire
    throw new Error("Quelque chose s'est mal passé");
    
    console.log("Ceci ne sera pas exécuté");
} catch (erreur) {
    // Gérer l'erreur
    console.error("Une erreur est survenue:", erreur.message);
    console.error("Stack trace:", erreur.stack);
} finally {
    // S'exécute toujours, qu'il y ait erreur ou non
    console.log("Bloc finally exécuté");
}

🔧 18.3 Créer des classes d'erreur personnalisées

// Créer une classe d'erreur personnalisée
class ValidationError extends Error {
    constructor(message, champ) {
        super(message);
        this.name = "ValidationError";
        this.champ = champ;
    }
}

class DatabaseError extends Error {
    constructor(message, code) {
        super(message);
        this.name = "DatabaseError";
        this.code = code;
    }
}

function validerAge(age) {
    if (age < 0) {
        throw new ValidationError("L'âge ne peut pas être négatif", "age");
    }
    if (age > 150) {
        throw new ValidationError("L'âge semble trop élevé", "age");
    }
    return true;
}

try {
    validerAge(-5);
} catch (erreur) {
    if (erreur instanceof ValidationError) {
        console.log(`Erreur de validation sur ${erreur.champ}: ${erreur.message}`);
    } else {
        console.log("Erreur inconnue:", erreur);
    }
}

⚡ Fonctions avancées

Closures, currying, IIFE, higher-order functions

🔒 19.1 Les closures

Une closure est une fonction qui se souvient de son environnement lexical même après que la fonction parente a fini de s'exécuter.

function creerCompteur() {
    let compteur = 0; // Variable "privée"
    
    return function() {
        compteur++;
        return compteur;
    };
}

const compteur1 = creerCompteur();
const compteur2 = creerCompteur();

console.log(compteur1()); // 1
console.log(compteur1()); // 2
console.log(compteur1()); // 3
console.log(compteur2()); // 1 (indépendant)
console.log(compteur2()); // 2

// Exemple pratique : fabrique de fonctions
function multiplier(facteur) {
    return function(nombre) {
        return nombre * facteur;
    };
}

const double = multiplier(2);
const triple = multiplier(3);

console.log(double(5)); // 10
console.log(triple(5)); // 15

🎯 19.2 Currying

Le currying transforme une fonction qui prend plusieurs arguments en une série de fonctions qui prennent un argument à la fois.

// Fonction normale
function addition(a, b, c) {
    return a + b + c;
}

// Version curryfiée
function additionCurry(a) {
    return function(b) {
        return function(c) {
            return a + b + c;
        };
    };
}

console.log(addition(1, 2, 3)); // 6
console.log(additionCurry(1)(2)(3)); // 6

// Version fléchée plus concise
const additionCurry2 = a => b => c => a + b + c;

// Utilisation pratique : pré-configurer des fonctions
const avecTaxe = taux => prix => prix * (1 + taux);
const tva20 = avecTaxe(0.20);
const tva5 = avecTaxe(0.05);

console.log(tva20(100)); // 120
console.log(tva5(100));  // 105

⚡ 19.3 IIFE

Une IIFE est une fonction qui s'exécute immédiatement après sa définition. Utile pour créer un scope privé.

// Syntaxe
(function() {
    console.log("IIFE classique");
})();

(function(nom) {
    console.log(`Bonjour ${nom}`);
})("Jean");

// Avec fonctions fléchées
(() => {
    console.log("IIFE fléchée");
})();

// Utilité : créer un espace privé
const compteur = (() => {
    let count = 0; // Privé
    
    return {
        incrementer: () => ++count,
        decrementer: () => --count,
        valeur: () => count
    };
})();

console.log(compteur.valeur());      // 0
console.log(compteur.incrementer()); // 1
console.log(compteur.incrementer()); // 2
console.log(compteur.decrementer()); // 1
// console.log(compteur.count); // undefined (privé)

🔄 19.4 Higher-Order Functions

Une higher-order function est une fonction qui prend une fonction en paramètre ou retourne une fonction.

// map, filter, reduce sont des HOF (déjà vues)

// Exemple : fonction qui applique une opération plusieurs fois
function repeter(n, action) {
    for (let i = 0; i < n; i++) {
        action(i);
    }
}

repeter(3, console.log);

// Exemple : fonction qui crée un validateur
function creerValidateur(regle, message) {
    return function(valeur) {
        return regle(valeur) ? null : message;
    };
}
      
const estNonVide = creerValidateur(
    val => val && val.trim().length > 0,
    "Ce champ ne peut pas être vide"
);

const estEmail = creerValidateur(
    val => /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(val),
    "Email invalide"
);

console.log(estNonVide(""));        // "Ce champ ne peut pas être vide"
console.log(estNonVide("Jean"));    // null
console.log(estEmail("test"));      // "Email invalide"
console.log(estEmail("test@test.com")); // null

// Task.js
export class Task {
    constructor(titre, description = "", dateEcheance = null, priorite = "moyenne") {
        this.id = Date.now().toString() + Math.random().toString(36).substr(2, 5);
        this.titre = titre;
        this.description = description;
        this.dateEcheance = dateEcheance;
        this.priorite = priorite; // "haute", "moyenne", "basse"
        this.termine = false;
        this.dateCreation = new Date().toISOString();
        this.dateModification = this.dateCreation;
    }
    
    toggle() {
        this.termine = !this.termine;
        this.dateModification = new Date().toISOString();
    }
    
    mettreAJour(donnees) {
        if (donnees.titre) this.titre = donnees.titre;
        if (donnees.description !== undefined) this.description = donnees.description;
        if (donnees.dateEcheance) this.dateEcheance = donnees.dateEcheance;
        if (donnees.priorite) this.priorite = donnees.priorite;
        this.dateModification = new Date().toISOString();
    }
    
    toJSON() {
        return {
            id: this.id,
            titre: this.titre,
            description: this.description,
            dateEcheance: this.dateEcheance,
            priorite: this.priorite,
            termine: this.termine,
            dateCreation: this.dateCreation,
            dateModification: this.dateModification
        };
    }
    
    static fromJSON(data) {
        const task = new Task(data.titre);
        task.id = data.id;
        task.description = data.description;
        task.dateEcheance = data.dateEcheance;
        task.priorite = data.priorite;
        task.termine = data.termine;
        task.dateCreation = data.dateCreation;
        task.dateModification = data.dateModification;
        return task;
    }
}

💾 20.5 Module Storage.js

// Storage.js
import { Task } from './Task.js';

export class Storage {
    static STORAGE_KEY = "todo_app_tasks";
    
    static sauvegarder(tasks) {
        try {
            const tasksData = tasks.map(task => task.toJSON());
            localStorage.setItem(this.STORAGE_KEY, JSON.stringify(tasksData));
            return true;
        } catch (erreur) {
            console.error("Erreur lors de la sauvegarde:", erreur);
            return false;
        }
    }
    
    static charger() {
        try {
            const data = localStorage.getItem(this.STORAGE_KEY);
            if (!data) return [];
            
            const tasksData = JSON.parse(data);
            return tasksData.map(taskData => Task.fromJSON(taskData));
        } catch (erreur) {
            console.error("Erreur lors du chargement:", erreur);
            return [];
        }
    }
    
    static vider() {
        localStorage.removeItem(this.STORAGE_KEY);
    }
    
    static exporter() {
        const data = localStorage.getItem(this.STORAGE_KEY);
        const blob = new Blob([data], { type: "application/json" });
        const url = URL.createObjectURL(blob);
        
        const a = document.createElement("a");
        a.href = url;
        a.download = `todo-backup-${new Date().toISOString().slice(0, 10)}.json`;
        a.click();
        
        URL.revokeObjectURL(url);
    }
    
    static async importer(fichier) {
        return new Promise((resolve, reject) => {
            const reader = new FileReader();
            
            reader.onload = (e) => {
                try {
                    const data = JSON.parse(e.target.result);
                    const tasks = data.map(taskData => Task.fromJSON(taskData));
                    resolve(tasks);
                } catch (erreur) {
                    reject(erreur);
                }
            };
            
            reader.onerror = reject;
            reader.readAsText(fichier);
        });
    }
}

// TaskManager.js
import { Storage } from './Storage.js';
import { Task } from './Task.js';

export class TaskManager {
    constructor() {
        this.tasks = [];
        this.filtre = "toutes"; // toutes, actives, terminees
        this.recherche = "";
        this.observateurs = [];
    }
    
    // Charger les tâches
    async charger() {
        try {
            const tasks = Storage.charger();
            this.tasks = tasks;
            this.notifierObservateurs();
            return tasks;
        } catch (erreur) {
            console.error("Erreur chargement:", erreur);
            return [];
        }
    }
    
    // Sauvegarder
    sauvegarder() {
        return Storage.sauvegarder(this.tasks);
    }
    
    // Ajouter une tâche
    ajouter(titre, description = "", dateEcheance = null, priorite = "moyenne") {
        const nouvelleTask = new Task(titre, description, dateEcheance, priorite);
        this.tasks.push(nouvelleTask);
        this.sauvegarder();
        this.notifierObservateurs();
        return nouvelleTask;
    }
    
    // Supprimer une tâche
    supprimer(id) {
        this.tasks = this.tasks.filter(task => task.id !== id);
        this.sauvegarder();
        this.notifierObservateurs();
    }
    
    // Modifier une tâche
    modifier(id, donnees) {
        const task = this.tasks.find(t => t.id === id);

            task.mettreAJour(donnees);
            this.sauvegarder();
            this.notifierObservateurs();
        }
    }

    toggleTermine(id) {
        const task = this.tasks.find(t => t.id === id);
        if (task) {
            task.toggle();
            this.sauvegarder();
            this.notifierObservateurs();
        }
    }

        // Filtrer par recherche
        if (this.recherche) {
            const searchLower = this.recherche.toLowerCase();
            resultat = resultat.filter(task => 
                task.titre.toLowerCase().includes(searchLower) ||
                task.description.toLowerCase().includes(searchLower)
            );
        }
        
        return resultat;
    }
    
    // Observer pattern pour UI
    ajouterObservateur(observateur) {
        this.observateurs.push(observateur);
    }
    
    notifierObservateurs() {
        const tachesFiltrees = this.getTachesFiltrees();
        this.observateurs.forEach(obs => obs(tachesFiltrees, this.tasks));
    }
    
    // Statistiques
    getStatistiques() {
        const total = this.tasks.length;
        const terminees = this.tasks.filter(t => t.termine).length;
        const actives = total - terminees;
        
        return {
            total,
            terminees,
            actives,
            pourcentage: total ? Math.round((terminees / total) * 100) : 0
        };
    }
}

// UI.js
export class UI {
    constructor(taskManager) {
        this.taskManager = taskManager;
        this.elements = this.capturerElements();
        this.initialiserEvenements();
        this.taskManager.ajouterObservateur(this.afficher.bind(this));
    }
    
    capturerElements() {
        return {
            taskList: document.getElementById("task-list"),
            taskForm: document.getElementById("task-form"),
            taskInput: document.getElementById("task-input"),
            descInput: document.getElementById("desc-input"),
            filterSelect: document.getElementById("filter-select"),
            searchInput: document.getElementById("search-input"),
            statsContainer: document.getElementById("stats-container"),
            exportBtn: document.getElementById("export-btn"),
            importBtn: document.getElementById("import-btn"),
            importFile: document.getElementById("import-file")
        };
    }
    
    initialiserEvenements() {
        // Formulaire d'ajout
        this.elements.taskForm.addEventListener("submit", (e) => {
            e.preventDefault();
            const titre = this.elements.taskInput.value.trim();
            if (titre) {
                this.taskManager.ajouter(titre, this.elements.descInput.value);
                this.elements.taskForm.reset();
            }
        });
        
        // Filtre
        this.elements.filterSelect.addEventListener("change", (e) => {
            this.taskManager.filtre = e.target.value;
            this.taskManager.notifierObservateurs();
        });
        
        // Recherche (debounce)
        let timeout;
        this.elements.searchInput.addEventListener("input", (e) => {
            clearTimeout(timeout);
            timeout = setTimeout(() => {
                this.taskManager.recherche = e.target.value;
                this.taskManager.notifierObservateurs();
            }, 300);
        });
        
        // Export
        this.elements.exportBtn.addEventListener("click", () => {
            Storage.exporter();
        });
        
        // Import
        this.elements.importBtn.addEventListener("click", () => {
            this.elements.importFile.click();
        });
        
        this.elements.importFile.addEventListener("change", async (e) => {
            const fichier = e.target.files[0];
            if (!fichier) return;
            
            try {
                const tasks = await Storage.importer(fichier);
                // Mettre à jour...
            } catch (erreur) {
                alert("Erreur lors de l'import: " + erreur.message);
            }
            
            this.elements.importFile.value = "";
        });
    }
    
    afficher(tachesFiltrees, toutesTaches) {
        this.afficherListe(tachesFiltrees);
        this.afficherStats(toutesTaches);
    }
    
    afficherListe(tasks) {
        this.elements.taskList.innerHTML = tasks.map(task => `
            <li class="task-item ${task.termine ? 'termine' : ''}" data-id="${task.id}">
                <input type="checkbox" class="task-checkbox" ${task.termine ? 'checked' : ''}>
                <div class="task-content">
                    <h3 class="task-title">${this.escapeHtml(task.titre)}</h3>
                    <p class="task-desc">${this.escapeHtml(task.description)}</p>
                    <div class="task-meta">
                        <span class="task-priority ${task.priorite}">${task.priorite}</span>
                        ${task.dateEcheance ? `<span class="task-date">📅 ${new Date(task.dateEcheance).toLocaleDateString()}</span>` : ''}
                    </div>
                </div>
                <div class="task-actions">
                    <button class="edit-btn" title="Modifier">✏️</button>
                    <button class="delete-btn" title="Supprimer">🗑️</button>
                </div>
            </li>
        `).join("");
        
        // Ajouter les événements sur les nouveaux éléments
        this.ajouterEvenementsTaches();
    }
    
    afficherStats(taches) {
        const total = taches.length;
        const terminees = taches.filter(t => t.termine).length;
        const pourcentage = total ? Math.round((terminees / total) * 100) : 0;
        
        this.elements.statsContainer.innerHTML = `
            <div class="stat-card">
                <span class="stat-value">${total}</span>
                <span class="stat-label">Total</span>
            </div>
            <div class="stat-card">
                <span class="stat-value">${terminees}</span>
                <span class="stat-label">Terminées</span>
            </div>
            <div class="stat-card">
                <span class="stat-value">${total - terminees}</span>
                <span class="stat-label">Actives</span>
            </div>
            <div class="stat-card">
                <span class="stat-value">${pourcentage}%</span>
                <span class="stat-label">Progression</span>
            </div>
        `;
    }
    
    ajouterEvenementsTaches() {
        // Checkboxes (toggle)
        document.querySelectorAll(".task-checkbox").forEach(checkbox => {
            checkbox.addEventListener("change", (e) => {
                const li = e.target.closest(".task-item");
                const id = li.dataset.id;
                this.taskManager.toggleTermine(id);
            });
        });
        
        // Boutons supprimer
        document.querySelectorAll(".delete-btn").forEach(btn => {
            btn.addEventListener("click", (e) => {
                const li = e.target.closest(".task-item");
                const id = li.dataset.id;
                if (confirm("Supprimer cette tâche ?")) {
                    this.taskManager.supprimer(id);
                }
            });
        });
        
        // Boutons modifier
        document.querySelectorAll(".edit-btn").forEach(btn => {
            btn.addEventListener("click", (e) => {
                const li = e.target.closest(".task-item");
                const id = li.dataset.id;
                const task = this.taskManager.tasks.find(t => t.id === id);
                if (task) {
                    const nouveauTitre = prompt("Modifier le titre", task.titre);
                    if (nouveauTitre && nouveauTitre.trim()) {
                        this.taskManager.modifier(id, { titre: nouveauTitre.trim() });
                    }
                }
            });
        });
    }
    
    escapeHtml(texte) {
        const div = document.createElement("div");
        div.textContent = texte;
        return div.innerHTML;
    }
}
